# Linguagem Uzumaki
Compilador com palavras reservadas inspiradas em palavras e frases do anime japonês Naruto

## Palavras Reservadas

      if = 'MADARA'
      else if = 'UCHIHA'
      else = 'OBITO'
      and = 'HASHIRAMA'
      or = 'TOBIRAMA'
      not = 'KABUTO'
      declaração de função = 'DATTEBAYO'
      for = 'KAGEBUNSHIN'
      to (repetir o for até ...) = 'NO'
      then(faça isso no laço) = 'JUTSU'
      step(incremento do for) = 'KARA'
      while = 'BUNSHIN'
